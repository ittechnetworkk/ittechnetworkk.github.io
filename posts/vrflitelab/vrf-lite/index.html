R3(config-if)#int e0/3

R3(config-if)#no sh
R3(config-if)#exit

#OSPF
R3(config)#router ospf 1 vrf Customer-A

R3(config-router)#network 1.1.5.0 0.0.0.3 area 0
R3(config-router)#capability vrf-lite 
R3(config-router)#exit
